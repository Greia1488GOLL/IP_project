```python
import logging  
import requests  
from datetime import datetime  
from telegram import Update, InlineKeyboardButton, InlineKeyboardMarkup  
from telegram.ext import Application, CommandHandler, MessageHandler, filters, ContextTypes, CallbackQueryHandler  
  
# Настройка  
logging.basicConfig(format='%(asctime)s - %(name)s - %(levelname)s - %(message)s', level=logging.INFO)  
logger = logging.getLogger(__name__)  
  
# Конфигурация  
TELEGRAM_TOKEN = '8541522268:AAGCXFjkAOq5zQusZC7vUrw-B4uXpB4LIK8'  
EODHD_API_KEY = '69940b504bb1f8.41159956'  
  
# База данных европейских акций  
STOCKS = {  
    'SAP.DE': 'SAP (Германия)', 'SIE.DE': 'Siemens (Германия)', 'VOW3.DE': 'Volkswagen (Германия)',  
    'BMW.DE': 'BMW (Германия)', 'AIR.PA': 'Airbus (Франция)', 'MC.PA': 'LVMH (Франция)',  
    'OR.PA': 'L\'Oreal (Франция)', 'ULVR.L': 'Unilever (UK)', 'SHEL.L': 'Shell (UK)',  
    'BP.L': 'BP (UK)', 'NESN.SW': 'Nestle (Швейцария)', 'ROG.SW': 'Roche (Швейцария)',  
    'ASML.AS': 'ASML (Нидерланды)', 'ENI.MI': 'Eni (Италия)', 'SAN.MC': 'Banco Santander (Испания)'  
}  
  
  
async def start(update: Update, context: ContextTypes.DEFAULT_TYPE):  
    keyboard = [  
        [InlineKeyboardButton("📈 Акции", callback_data='stocks')],  
        [InlineKeyboardButton("🔍 Поиск", callback_data='search'),  
         InlineKeyboardButton("❓ Помощь", callback_data='help')]  
    ]  
    await update.message.reply_text("🇪🇺 Европейские акции\nВыберите действие:",  
                                    reply_markup=InlineKeyboardMarkup(keyboard))  
  
  
async def stocks_list(update: Update, context: ContextTypes.DEFAULT_TYPE):  
    keyboard = []  
    for ticker, name in list(STOCKS.items())[:8]:  
        keyboard.append([InlineKeyboardButton(name, callback_data=f'price_{ticker}')])  
  
    query = update.callback_query  
    await query.answer()  
    await query.edit_message_text("Выберите акцию:", reply_markup=InlineKeyboardMarkup(keyboard))  
  
  
async def get_price(update: Update, context: ContextTypes.DEFAULT_TYPE):  
    ticker = context.args[0].upper() if context.args else update.callback_query.data.replace('price_', '')  
  
    try:  
        url = f"https://eodhd.com/api/real-time/{ticker}"  
        response = requests.get(url, params={"api_token": EODHD_API_KEY, "fmt": "json"}).json()  
  
        if response:  
            name = STOCKS.get(ticker, ticker)  
            price = response.get('close', 'N/A')  
            change = response.get('change_p', 'N/A')  
  
            msg = f"📊 *{name}*\n`{ticker}`\n💰 {price} EUR\n📊 {change}%\n🕐 {datetime.now().strftime('%H:%M')}"  
  
            keyboard = [[InlineKeyboardButton("🔄 Обновить", callback_data=f'price_{ticker}')]]  
  
            if update.message:  
                await update.message.reply_text(msg, parse_mode='Markdown', reply_markup=InlineKeyboardMarkup(keyboard))  
            else:  
                await update.callback_query.edit_message_text(msg, parse_mode='Markdown',  
                                                              reply_markup=InlineKeyboardMarkup(keyboard))  
    except Exception as e:  
        await update.message.reply_text(f"❌ Ошибка: {ticker}")  
  
  
async def search(update: Update, context: ContextTypes.DEFAULT_TYPE):  
    query = ' '.join(context.args).lower() if context.args else ''  
    if not query:  
        await update.message.reply_text("Используйте: /search название")  
        return  
  
    results = [f"• {name} - `{ticker}`" for ticker, name in STOCKS.items() if query in name.lower()]  
    await update.message.reply_text("\n".join(results[:5]) or "❌ Не найдено", parse_mode='Markdown')  
  
  
async def button_handler(update: Update, context: ContextTypes.DEFAULT_TYPE):  
    query = update.callback_query  
    await query.answer()  
  
    if query.data == 'stocks':  
        await stocks_list(update, context)  
    elif query.data == 'search':  
        await query.edit_message_text("Введите /search название_компании")  
    elif query.data == 'help':  
        await query.edit_message_text("/stocks - список акций\n/search название - поиск\n/price ТИКЕР - цена")  
    elif query.data.startswith('price_'):  
        context.args = [query.data.replace('price_', '')]  
        await get_price(update, context)  
  
  
def main():  
    app = Application.builder().token(TELEGRAM_TOKEN).build()  
    app.add_handler(CommandHandler("start", start))  
    app.add_handler(CommandHandler("stocks", stocks_list))  
    app.add_handler(CommandHandler("price", get_price))  
    app.add_handler(CommandHandler("search", search))  
    app.add_handler(CallbackQueryHandler(button_handler))  
    app.run_polling()  
  
  
if __name__ == '__main__':  
    main()
```
