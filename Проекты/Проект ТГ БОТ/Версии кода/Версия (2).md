async def get_price(update: Update, context: ContextTypes.DEFAULT_TYPE):
    if update.callback_query:
        ticker = update.callback_query.data.replace('price_', '')
    else:
        ticker = context.args[0].upper() if context.args else None
    
    if not ticker:
        return
    
    try:
        url = f"https://eodhd.com/api/real-time/{ticker}"
        response = requests.get(url, params={"api_token": EODHD_API_KEY, "fmt": "json"}).json()
        
        if response:
            info = STOCKS.get(ticker, {'name': ticker, 'country': 'Неизвестно', 'icon': '🏢'})
            price = response.get('close', 0)
            prev_close = response.get('previousClose', price)
            change_abs = price - prev_close
            change_pct = (change_abs / prev_close * 100) if prev_close else 0
            volume = response.get('volume', 'N/A')
            currency = 'RUB' if '.ME' in ticker else 'EUR'
            
            trend = "📈" if change_abs >= 0 else "📉"
            
            # Правильное формирование строки с флагом страны
            if info['country'] == 'РФ':
                country_text = "🇷🇺 Россия"
            else:
                country_text = f"🇪🇺 {info['country']}"
            
            msg = (
                f"{info['icon']} *{info['name']}* ({ticker})\n"
                f"{country_text}\n"
                f"💰 {price:,.2f} {currency}\n"
                f"{trend} {change_abs:+.2f} {currency} ({change_pct:+.2f}%)\n"
                f"📊 Объём: {volume:,}\n"
                f"🕐 {datetime.now().strftime('%d.%m.%Y %H:%M')}"
            )
            
            # Определяем текст кнопки избранного
            user_id = update.effective_user.id
            is_favorite = ticker in user_favorites.get(user_id, [])
            fav_button_text = "⭐ Удалить" if is_favorite else "⭐ Добавить"
            
            keyboard = [
                [InlineKeyboardButton("🔄 Обновить", callback_data=f'price_{ticker}'),
                 InlineKeyboardButton("📈 График", callback_data=f'chart_{ticker}')],
                [InlineKeyboardButton(fav_button_text, callback_data=f'fav_{ticker}'),
                 InlineKeyboardButton("🔙 Назад", callback_data='main_menu')]
            ]
            
            if update.callback_query:
                await update.callback_query.edit_message_text(msg, parse_mode='Markdown',
                                                              reply_markup=InlineKeyboardMarkup(keyboard))
            else:
                await update.message.reply_text(msg, parse_mode='Markdown',
                                                reply_markup=InlineKeyboardMarkup(keyboard))
    except Exception as e:
        error_msg = f"❌ Ошибка получения данных для {ticker}"
        if update.callback_query:
            await update.callback_query.edit_message_text(error_msg)
        else:
            await update.message.reply_text(error_msg)

async def show_chart(update: Update, context: ContextTypes.DEFAULT_TYPE):
    query = update.callback_query
    await query.answer()
    
    ticker = query.data.replace('chart_', '')
    info = STOCKS.get(ticker, {'name': ticker})
    
    try:
        end_date = datetime.now()
        start_date = end_date - timedelta(days=30)
        
        url = f"https://eodhd.com/api/eod/{ticker}"
        params = {
            "api_token": EODHD_API_KEY,
            "fmt": "json",
            "from": start_date.strftime('%Y-%m-%d'),
            "to": end_date.strftime('%Y-%m-%d')
        }
        
        response = requests.get(url, params=params).json()
        
        if response and len(response) > 1:
            dates = [datetime.strptime(item['date'], '%Y-%m-%d') for item in response]
            closes = [item['close'] for item in response]
            
            plt.figure(figsize=(10, 6))
            plt.plot(dates, closes, 'b-', linewidth=2)
            plt.title(f'{info["name"]} - цена за 30 дней')
            plt.xlabel('Дата')
            plt.ylabel('Цена')
            plt.grid(True, alpha=0.3)
            plt.xticks(rotation=45)
            plt.tight_layout()
            
            buf = BytesIO()
            plt.savefig(buf, format='png')
            buf.seek(0)
            plt.close()
            
            keyboard = [[InlineKeyboardButton("🔙 Назад", callback_data=f'price_{ticker}')]]
            await query.message.reply_photo(photo=buf, caption=f"📊 График {info['name']}",
                                           reply_markup=InlineKeyboardMarkup(keyboard))
        else:
            await query.edit_message_text("❌ Недостаточно данных для графика")
            
    except Exception as e:
        await query.edit_message_text("❌ Ошибка построения графика")

async def handle_favorite(update: Update, context: ContextTypes.DEFAULT_TYPE):
    query = update.callback_query
    await query.answer()
    
    user_id = query.from_user.id
    ticker = query.data.replace('fav_', '')
    
    if user_id not in user_favorites:
        user_favorites[user_id] = []
    
    if ticker not in user_favorites[user_id]:
        user_favorites[user_id].append(ticker)
        await query.answer(f"⭐ {STOCKS[ticker]['name']} добавлена в избранное")
    else:
        user_favorites[user_id].remove(ticker)
        await query.answer(f"⭐ {STOCKS[ticker]['name']} удалена из избранного")
    
    # Возвращаемся к цене (обновляем сообщение)
    context.args = [ticker]
    await get_price(update, context)

async def get_top_movers(update: Update, context: ContextTypes.DEFAULT_TYPE, movers_type='gainers'):
    query = update.callback_query
    await query.answer()
    
    await query.edit_message_text("🔄 Загружаем данные...")
    
    movers = []
    for ticker in STOCKS.keys():
        try:
            url = f"https://eodhd.com/api/real-time/{ticker}"
            response = requests.get(url, params={"api_token": EODHD_API_KEY, "fmt": "json"}).json()
            
            if response:
                prev_close = response.get('previousClose', 0)
                price = response.get('close', 0)
                if prev_close:
                    change_pct = ((price - prev_close) / prev_close) * 100
                    movers.append((ticker, change_pct, price))
        except:
            continue
    
    movers.sort(key=lambda x: x[1], reverse=(movers_type == 'gainers'))
    top_movers = movers[:10]
    
    msg = f"📈 Топ роста\n\n" if movers_type == 'gainers' else f"📉 Топ падения\n\n"
    
    for ticker, change, price in top_movers:
        info = STOCKS[ticker]
        trend = "🟢" if change > 0 else "🔴"
        msg += f"{trend} {info['icon']} {info['name']}\n{change:+.2f}% ({price:.2f})\n\n"
    
    keyboard = [[InlineKeyboardButton("🔙 Назад", callback_data='main_menu')]]
    await query.edit_message_text(msg, reply_markup=InlineKeyboardMarkup(keyboard))

async def search(update: Update, context: ContextTypes.DEFAULT_TYPE):
    query = ' '.join(context.args).lower() if context.args else ''
    
    if not query:
        await update.message.reply_text("Используйте: /search название_компании")
        return
    
    results = []
    for ticker, info in STOCKS.items():
        if query in info['name'].lower() or query in info['country'].lower():
            results.append(f"• {info['icon']} {info['name']} - `{ticker}`")
    
    await update.message.reply_text("\n".join(results[:10]) or "❌ Ничего не найдено", 
                                    parse_mode='Markdown')

async def button_handler(update: Update, context: ContextTypes.DEFAULT_TYPE):
    query = update.callback_query
    await query.answer()
    
    if query.data == 'main_menu':
        await show_main_menu(update, context, edit_message=True)
    elif query.data == 'stocks_eu':
        await stocks_list(update, context, 'EU')
    elif query.data == 'stocks_ru':
        await stocks_list(update, context, 'RU')
    elif query.data == 'favorites':
        await show_favorites(update, context)
    elif query.data == 'search':
        await query.edit_message_text("Введите /search название_компании")
    elif query.data == 'top_gainers':
        await get_top_movers(update, context, 'gainers')
    elif query.data == 'top_losers':
        await get_top_movers(update, context, 'losers')
    elif query.data == 'help':
        help_text = (
            "📚 Доступные команды:\n\n"
            "/start - Главное меню\n"
            "/stocks - Список акций\n"
            "/search название - Поиск\n"
            "/price ТИКЕР - Цена акции\n\n"
            "⭐ Избранное - сохраняйте понравившиеся акции\n"
            "📈 Графики - динамика за 30 дней\n"
            "🇪🇺/🇷🇺 - фильтр по региону"
        )
        await query.edit_message_text(help_text, 
                                      reply_markup=InlineKeyboardMarkup([[
                                          InlineKeyboardButton("🔙 Назад", callback_data='main_menu')
                                      ]]))
    elif query.data.startswith('price_'):
        await get_price(update, context)
    elif query.data.startswith('chart_'):
        await show_chart(update, context)
    elif query.data.startswith('fav_'):
        await handle_favorite(update, context)

def main():
    app = Application.builder().token(TELEGRAM_TOKEN).build()
    
    app.add_handler(CommandHandler("start", start))
    app.add_handler(CommandHandler("stocks", lambda u, c: stocks_list(u, c, 'all')))
    app.add_handler(CommandHandler("price", get_price))
    app.add_handler(CommandHandler("search", search))
    
    app.add_handler(CallbackQueryHandler(button_handler))
    
    print("✅ Бот запущен!")
    app.run_polling()

if __name__ == '__main__':
    main()
